272 void c64::interruptIRQ(bool state)
273 {
274  if (state)
275  {
276  if (irqCount == 0)
277  cpu.triggerIRQ();
278 
279  irqCount ++;
280  }
281  else
282  {
283  irqCount --;
284  if (irqCount == 0)
285  cpu.clearIRQ();
286  }
